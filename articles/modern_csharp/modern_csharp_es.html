        public static object CreateShape(string shapeDescription)
        {
            switch (shapeDescription)
            {
                case "circle":
                    return (2, "circle");

                case "square":
                    return (4, "square");

                case "large-circle":
                    return (12, "large-circle");

                case string o when (o.Trim()?.Length ?? 0) == 0:
                    // white space
                    return null;
                case null:
                    return null;
                default:
                    return "invalid shape description";
            }
        }
